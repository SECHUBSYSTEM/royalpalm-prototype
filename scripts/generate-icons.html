PWA Icon Generator

Open browser console and run the code below to generate icons:

// Generate 192x192 icon
const canvas192 = document.createElement('canvas');
canvas192.width = 192;
canvas192.height = 192;
const ctx192 = canvas192.getContext('2d');
ctx192.fillStyle = '#2c3e50';
ctx192.fillRect(0, 0, 192, 192);
ctx192.fillStyle = '#ffffff';
ctx192.font = 'bold 64px Arial';
ctx192.textAlign = 'center';
ctx192.textBaseline = 'middle';
ctx192.fillText('RP', 96, 96);
const link192 = document.createElement('a');
link192.download = 'icon-192.png';
link192.href = canvas192.toDataURL();
link192.click();
